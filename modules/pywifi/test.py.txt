import pywifi
from pywifi import const
 
def list_pywifi_constants():
    # 获取 const 模块中的所有属性
    all_attributes = dir(const)
    
    # 过滤出常量（通常是大写的属性名）
    constants = [attr for attr in all_attributes if attr.isupper()]
    
    # 打印每个常量及其值
    for constant in constants:
        print(f"{constant}: {getattr(const, constant)}")

    for constant in constants:
        if "AUTH" in constant:
        	print(f"\t{constant}: {getattr(const, constant)}")
 
if __name__ == "__main__":
    list_pywifi_constants()
